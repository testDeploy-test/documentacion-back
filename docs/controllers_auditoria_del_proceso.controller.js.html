/**
 * Controller to obtain all `auditoria_del_proceso` records.
 * @function getAllProcesses
 * @async
 * @param {any} req - The `req` parameter represents the HTTP request object, which contains 
 * information about the incoming request such as headers, parameters, and body
 * data. It is commonly used to access data sent by the client to the server.
 * @param {any} res - The `res` parameter in the code snippet refers to the response object in Node.js. It is
 * used to send a response back to the client making the request. In this case, the `res` object is
 * used to send a JSON response containing the data fetched from the database or an error message.
 * @returns {Promise<Array<Object>>} The `getAllProcesses` function is returning a list of all processes with certain attributes
 * excluded, including related audit format data. The processes are ordered by `IDproceso` in ascending
 * order. If successful, it returns a JSON response with the list of processes. If there is an error,
 * it logs the error and returns a 500 status with an error message indicating an internal server
 * error.
 */
export const getAllProcesses = async (req, res) => {
    try {
        // Try to get all `auditoria_del_proceso` records, sorted by IDproceso in ascending order.
        const allProcesses = await process.findAll({ 
            attributes: { exclude: ["IDauditoria"] },
            include: {
                model: audit_format,
                attributes: [ "IDauditoria", "PO" ]
            },
            order: [[ "IDproceso", "ASC" ]]
        });
        // If the query is successful, returns the `auditoria_del_proceso` records as a JSON response.
        return res.json(allProcesses);
    } catch (error) {
        // If there is an error in the query, it prints an error message to the console and return an error message.
        console.error(error);
        return res.status(500).json({ error: "There was an internal server error." });
    }
}

export const addProcess = async (req, res) => {
    // Declare the variable `imagen`.
    let imagen;
    // If does not exist req.files, the variable `imagen` will be req.body.imagen 
    if(!req.files) {
        imagen = req.body.imagen

        imagen = await uploadImage(req.files.file)

        pnc,
        fecha_apertura,
        cantidad_rechazada,
        defectos_encontrados,
        operario,
        accion,

        IDauditoria
    } = req.body;
    try {
        // Attempt to create a new `auditoria_del_proceso` record with the provided data.
        const newProcess = await process.create({
            pnc,
            fecha_apertura,
            cantidad_rechazada,
            defectos_encontrados,
            operario,
            accion,
            aprobado_por,
            comentario_accion,
            cantidad_descarte,
            cantidad_aceptada,

            fecha_cierre,

/**
 * Controller to update a record by IDproceso of `auditoria_del_proceso`.
 * @function updateProcess
 * @async
 * @param {any} req - The `req` parameter represents the HTTP request object, which contains 
 * information about the incoming request such as headers, parameters, and body
 * data. It is commonly used to access data sent by the client to the server.
 * @param {any} res - The `res` parameter in the code snippet refers to the response object in Node.js. It is
 * used to send a response back to the client making the request. In this case, the `res` object is
 * used to send a JSON response containing the data fetched from the database or an error message.
 * @returns {Promise<Array<number>>} The `updateProcess` function returns a JSON response containing an array with the number of records affected
 * if the update is successful. If there is an error during the update process, it returns
 * a 500 status with an error message indicating that there was an internal server error.
 */
export const updateProcess = async (req, res) => {
    // Declare the variable `data`
    let data;
    // If req.files does not exist, `data` will be only the req.body
    if(!req.files) {
        data = req.body;
    } else {
        // If req.files exist, `data` will be the req.body and `imagen` (path of the new image)
        let imagen = await uploadImage(req.files.file)
        const textFields = req.body;
        data = { textFields, imagen}
    }
    try {
        // Get the previous value of the record.
        const prevImg = await process.findByPk(req.params.IDproceso);
        // Attempt to update the record with the new data. 
        const uProcess = await process.update(data, { where: { IDproceso: req.params.IDproceso} });
        // Delete the previus image if data contains a new image.
        if(req.files && prevImg.imagen != "") {
            removeImage(prevImg.imagen)
        }
        // If the update is successful, returns an array with the number of affected rows as a JSON response.
        return res.json(uProcess)
    } catch (error) {
        // If there is an error in the update, prints an error message to the console and returns an error message.
        console.error(error);
        return res.status(500).json({ error: "There was an internal server error." });
    }
}

 * @returns {Promise<Object>} The `deleteProcess` function returns a JSON response with a success message if the record
 * is successfully deleted, or an error message if there is an issue during the deletion process.
 */
export const deleteProcess = async (req, res) => {
    try {
        // Get the previous value of the record.
        const object = await process.findByPk(req.params.IDproceso);
        // If the record does not exist, returns error 404 - This record does not exist.
        if(!object) {
            return res.stauts(404).json({ error: "This record does not exist." })
        }
        // If the field imagen is not null, delete the image.
        if(object.imagen != "") {
            removeImage(object.imagen)
        }
        // Attempts to delete the `auditoria_delproceso` record with the specifed IDproceso.
        await process.destroy({ where: { IDproceso: req.params.IDproceso } });
        // If the deletion is successful, a success message is returned.
        return res.status(200).json({ message: `Record ${req.params.IDproceso} deleted.`});

    } catch (error) {
        // If there is an error in the update, prints an error message to the console and returns an error message.
      console.error(error);
      return res.status(500).json({ error: "There was an internal server error." });
    }
}
