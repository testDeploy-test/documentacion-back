export const getAllDefects = async (req, res) => {
    try {
        // Try to get all `defectos` records, sorted by codigoDefecto in ascending order.
        const allDefects = await defects.findAll({
            order: [["codigoDefecto", "ASC"]]
        });
        // If the query is successful, returns the `defectos` records as a JSON response.
        return res.json(allDefects);
    } catch (error) {
        // If there is an error in the query, it prints an error message to the console and returns an error message.
        console.error(error);
        return res.status(500).json({ error: "There was an internal server error" });
    }
}
