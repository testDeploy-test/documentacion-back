import path from "path";
import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

/**
 * Uploads an image to the server and returns a promise with the image path.
 * @function uploadImage
 * @param {Object} image - The image object to upload. It must have a `name` property containing the 
 * name of the image file. 
 * @returns {Promise<string>} A promise that is resolved with the path to the uploaded image if the 
 * upload is successful, or rejected with an error if the upload fails.
 */
const uploadImage = (image) => {
    return new Promise((resolve, reject) => {
        // Get the file extension of the image file
        let ext = path.extname(image.name);
         // Create a unique file name
        const fileName = `${Date.now()}-${Math.round(Math.random() * 1E9)}${ext}`;

        // Create the path where the image will be saved, in production it will only be (“uploads”, fileName)
        const rute = path.join(__dirname, "..", "uploads", fileName);

        // Attempts to move the image to the specified path.
        image.mv(rute, error => {

            rejects the promise with an error */

}
