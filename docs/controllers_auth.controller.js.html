import users from "../models/usuarios.model.js";
import bcrypt from "bcryptjs";
import crateAccesToken from "../lib/jwt.js";
import jwt from "jsonwebtoken"
import { configDotenv } from "dotenv";

configDotenv();

/**
 * The `login` function in JavaScript handles user authentication by checking user credentials, time
 * restrictions, and generating a token for successful logins.
 * @function login
 * @async
 * @param {any} req - The `req` parameter represents the HTTP request object, which contains 
 * information about the incoming request such as headers, parameters, and body
 * data. It is commonly used to access data sent by the client to the server.
 * @param {any} res - The `res` parameter in the code snippet refers to the response object in Node.js. It is
 * used to send a response back to the client making the request. In this case, the `res` object is
 * used to send a JSON response containing the data fetched from the database or an error message.
 * @returns {Promise<Object>} The login function returns different responses based on the conditions met during the
 * execution:
 */
export const login = async (req, res) => {
    // Check if the usuario and contraseña exist in req.body, if not return an error message.
    if(!req.body.usuario || !req.body.contraseña) return res.status(400).json({ error: "User and password are required."})

    const { usuario, contraseña } = req.body;
    const hora_actual = new Date;
    const expirationDate = new Date;

    try {
        // Get the user 
        const user = await users.findOne({ where: { usuario: usuario, activo: true } });
        if(!user) {
            return res.status(404).json({ message: "User not found." });
        }
        // Declare the hour of beginning workday
        const firstHour = user.hora_inicio;
        // Declare the hour of end workday.
        const secondHour = user.hora_fin;
        // Separete the hours in hours, minutes and seconds.
        const hourParts1 = firstHour.split(":");
        const hourParts2 = secondHour.split(":");
        // Declare the constants in order to make the comparison between the current time and today's working hours
        const hora_inicio = new Date();
        const hora_fin = new Date();
        // Inserts the corresponding hours, minutes and seconds
        hora_inicio.setHours(parseInt(hourParts1[0]), parseInt(hourParts1[1]), parseInt(hourParts1[2]));
        hora_fin.setHours(parseInt(hourParts2[0]), parseInt(hourParts2[1]), parseInt(hourParts2[2]));
        expirationDate.setHours(parseInt(hourParts2[0]), parseInt(hourParts2[1]), parseInt(hourParts2[2]));

        // If the user wants to log in outside working hours, an error message is returned.
        if(hora_actual < hora_inicio || hora_actual > hora_fin) {
            return res.status(400).json({  message: "Wrong time." })
        }
        // Check if the passwords are equal.
        const isMatch = await bcrypt.compare(contraseña, user.contraseña);
        // If the password isn't equal, returns an error message. 
        if (!isMatch) { 
            return res.status(400).json({  message: "Incorrect password." })
        }
        
        // Create an access token.
        const token = await crateAccesToken({ id: user.IDusuario, isAdmin: user.superAdmin})
        res.cookie('token', token, { /*secure: true, sameSite: 'Strict',*/ expires: expirationDate, httpOnly: true });
        // If the login is successful, a success message is returned.
        return res.json({message: "Login successful.", id: user.IDusuario});
    } catch (error) {
        // If there is an error in the update, prints an error message to the console and returns an error message.
        console.error(error);
        return res.status(500).json({ error: 'There was an internal server error.' });
    }
}


/**
 * The `logout` function logs out a user by clearing the token cookie and returning a success message.
 * @function logout
 * @async
 * @param {any} req - The `req` parameter represents the HTTP request object, which contains 
 * information about the incoming request such as headers, parameters, and body
 * data. It is commonly used to access data sent by the client to the server.
 * @param {any} res - The `res` parameter in the code snippet refers to the response object in Node.js. It is
 * used to send a response back to the client making the request. In this case, the `res` object is
 * used to send a JSON response containing the data fetched from the database or an error message.
 * @returns {Promise<Object>} The `logout` function is returning a JSON response with a message indicating that the
 * logout was successful. The response structure looks like this:
 * { message: 'Logout successful.' }
 */
export const logout = async (req, res) => {
    try {
        // Clean the access token.
        res.cookie('token', null, { /*secure: true, sameSite: 'Strict',*/ expires: new Date(0), httpOnly: true });
        // If logout is successful, a success message is returned.
        return res.json({ message: 'Logout successful.' });
    } catch (error) {
        // If there is an error in the update, prints an error message to the console and returns an error message.
        console.error(error);
        return res.status(500).json({ error: 'There was an internal server error.' });
    }
}



/**
 * The function verifies a token, checks if it is valid, and returns an "Authorized" message if
 * successful.
 * @function verify
 * @async
 * @param {any} req - The `req` parameter represents the HTTP request object, which contains 
 * information about the incoming request such as headers, parameters, and body
 * data. It is commonly used to access data sent by the client to the server.
 * @param {any} res - The `res` parameter in the code snippet refers to the response object in Node.js. It is
 * used to send a response back to the client making the request. In this case, the `res` object is
 * used to send a JSON response containing the data fetched from the database or an error message.
 * @returns {Promise<Object>} The `verify` function is checking for the presence of a token in the request cookies. If
 * the token is not found, it returns a 401 status with an error message "Unauthorized".
 */
export const verify = async (req, res) => {
    // Extrct the token.
    const { token } = req.cookies;
    // If no token return error 401 - Unathorized.
    if(!token) return res.status(401).json({error: "Unauthorized"});

    try {
        // Check if the token is valid- 
        const decoded = jwt.verify(token, process.env.KEY );
        // If invalid token return error 401 - Unathorized.
        if(!decoded) return res.status(401).json({error: "Unauthorized"});

        // Get the user by id in the token.
        const user = await users.findByPk(decoded.id);
        // If user does not exist return error 401 - Unathorized.
        if(!user) return res.status(401).json({error: "Unauthorized"})
        
        // If token verification successful, a success message is returned.
        return res.json({ message: "Authorized"})
    } catch (error) {
        // If there is an error in the update, prints an error message to the console and returns an error message.
        console.error(error);
        return res.status(500).json({ error: "There was an internal server error."})
    }
}
